// Firebase Storage Security Rules
// Copy these rules to your Firebase Console → Storage → Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Allow public read access but authenticated write for specialty images
    match /specialties/{allPaths=**} {
      allow read: if true; // Public read access for images
      allow write: if request.auth != null; // Only authenticated users can upload
    }
    
    // Allow authenticated users to upload and read test files
    match /test/{allPaths=**} {
      allow read, write: if request.auth != null;
    }
    
    // Default rule - deny all other access
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
